Code Examples

Learn how to use catalogmx in your projects

Validate RFC

import { validateRfc, detectRfcType } from 'catalogmx';

// Validate an RFC
const isValid = validateRfc('GODE561231GR8');
console.log(isValid); // true

// Detect RFC type
const type = detectRfcType('GODE561231GR8');
console.log(type); // 'persona_fisica'

Generate RFC

import { generateRfcPersonaFisica } from 'catalogmx';

const rfc = generateRfcPersonaFisica({
  nombre: 'Juan',
  apellidoPaterno: 'Garcia',
  apellidoMaterno: 'Lopez',
  fechaNacimiento: new Date(1990, 4, 15)
});
console.log(rfc); // e.g., GALJ900515XXX

Validate CURP

import { validateCurp, CURPValidator } from 'catalogmx';

// Simple validation
const isValid = validateCurp('GODE561231HDFRRL09');

// Detailed validation with info extraction
const validator = new CURPValidator('GODE561231HDFRRL09');
if (validator.isValid()) {
  console.log(validator.getBirthDate());
  console.log(validator.getGender());
  console.log(validator.getState());
}

Validate CLABE

import { validateClabe, CLABEValidator } from 'catalogmx';

// Quick validation
const isValid = validateClabe('002010077777777771');

// Extract bank info
const clabe = new CLABEValidator('002010077777777771');
if (clabe.isValid()) {
  console.log(clabe.getBankCode());   // '002'
  console.log(clabe.getBankName());   // 'BANAMEX'
  console.log(clabe.getPlazaCode()); // '010'
}

Access Bank Catalog

import { BankCatalog } from 'catalogmx';

// Get all banks
const banks = BankCatalog.getAll();

// Get bank by code
const banamex = BankCatalog.getBankByCode('002');
console.log(banamex.name); // 'BANAMEX'

// Get only SPEI-enabled banks
const speiBanks = BankCatalog.getSPEIBanks();

// Search banks
const results = BankCatalog.searchBanks('BBVA');
console.log(results.length); // Banks matching 'BBVA'

Access State Catalog

import { StateCatalog } from 'catalogmx';

// Get all states
const states = StateCatalog.getAll();
console.log(states.length); // 32

// Get state by CURP code
const state = StateCatalog.getByCode('DF');
console.log(state.name); // 'CIUDAD DE MEXICO'

// Get by INEGI code
const cdmx = StateCatalog.getByINEGICode('09');
console.log(cdmx.abbreviation); // 'CDMX'

Postal Code Lookup

import { CodigosPostales } from 'catalogmx';

// Get localities by postal code
const localities = CodigosPostales.getByPostalCode('06600');
console.log(localities);
// [{ cp: '06600', colonia: 'Roma Norte', ... }]

// Get all postal codes for a state
const cdmxCodes = CodigosPostales.getByState('Ciudad de Mexico');

// Search by colony name
const results = CodigosPostales.searchByColony('Condesa');

SAT CFDI Catalogs

import {
  RegimenFiscalCatalog,
  UsoCFDICatalog,
  FormaPagoCatalog
} from 'catalogmx';

// Tax regimes
const regimes = RegimenFiscalCatalog.getAll();
const regime = RegimenFiscalCatalog.getByCode('601');
console.log(regime.description);
// 'General de Ley Personas Morales'

// Invoice usage
const uso = UsoCFDICatalog.getByCode('G03');
console.log(uso.description); // 'Gastos en general'

// Payment methods
const formas = FormaPagoCatalog.getAll();
console.log(formas); // Cash, Transfer, Card, etc.

ISR Calculation

import { ISRCalculator } from 'catalogmx';

const result = ISRCalculator.calcular({
  ingresoGravable: 50000,
  periodo: 'mensual',
  anio: 2024
});

console.log(result.isrRetenido);    // Tax amount
console.log(result.tasaEfectiva);   // Effective rate
console.log(result.limiteInferior); // Lower bracket
console.log(result.cuotaFija);      // Fixed fee

Validate NSS

import { validateNss, NSSValidator } from 'catalogmx';

// Quick validation
const isValid = validateNss('12345678901');

// Detailed validation
const nss = new NSSValidator('12345678901');
if (nss.isValid()) {
  console.log(nss.getSubdelegation());
  console.log(nss.getBirthYear());
  console.log(nss.getRegistrationYear());
}

Currency Catalog

import { MonedasDivisas } from 'catalogmx';

// Get all currencies
const currencies = MonedasDivisas.getAll();

// Get currency by ISO code
const mxn = MonedasDivisas.getByCode('MXN');
console.log(mxn.name); // 'Peso Mexicano'
console.log(mxn.decimals); // 2

// Get USD
const usd = MonedasDivisas.getByCode('USD');
console.log(usd.name); // 'Dolar Estadounidense'

Municipalities & Localities

import {
  MunicipiosCatalog,
  LocalidadesCatalog
} from 'catalogmx';

// Get municipalities by state
const muns = MunicipiosCatalog.getByState('09');
console.log(muns.length); // 16 (CDMX)

// Get localities with GPS coordinates
const localities = LocalidadesCatalog.getByMunicipality(
  '09', // State
  '015' // Municipality (Cuauhtemoc)
);

// Each locality includes:
// - latitude, longitude, altitude
// - population data
// - urban/rural classification

Installation

Get started with catalogmx in your project

TypeScript / JavaScript

# Using npm
npm install catalogmx

# Using yarn
yarn add catalogmx

# Using pnpm
pnpm add catalogmx
// ES Modules
import { validateRfc, BankCatalog } from 'catalogmx';

// CommonJS
const { validateRfc, BankCatalog } = require('catalogmx');

Python

# Using pip
pip install catalogmx

# Using uv (10-100x faster)
uv pip install catalogmx
from catalogmx import validate_rfc, BankCatalog

# Validate RFC
is_valid = validate_rfc("GODE561231GR8")

# Get banks
banks = BankCatalog.get_all()

Dart / Flutter

# pubspec.yaml
dependencies:
  catalogmx: ^0.3.0
import 'package:catalogmx/catalogmx.dart';

// Validate RFC
final isValid = validateRfc('GODE561231GR8');

// Get banks
final banks = BankCatalog.getAll();
